% Numbering
\setbeamertemplate{footline}{
    \leavevmode%
    \hbox{%
        \ifnum\value{framenumber}>1
            \hspace*{1em} 
            \usebeamercolor[fg]{page number in head/foot}
            \fontsize{10pt}{12pt}\selectfont
            \textbf{\insertframenumber}
        \fi
    }%
    \vskip2.35ex 
}

% Table of contents

\setbeamertemplate{section in toc}{%
    \leavevmode\leftskip=-2em%
    \hspace{5pt}
    \llap{\color{bulletcolour}\scalebox{0.7}{$\blacksquare$}}% Define a custom bullet point
    \hspace{2pt}
    {\large\usebeamercolor[fg]{frametitle}\inserttocsection}% Inserts the section title
}
\setbeamertemplate{subsection in toc}{%
    \vspace{3pt}%
    \leavevmode\leftskip=3em%
    % \hspace{25pt}%
    \llap{\color{bulletcolour}\scalebox{0.6}{$\blacksquare$}}% Define a custom bullet for subsections
    \hspace{5pt}%
    \inserttocsubsection%
    \par% Forces a new paragraph, ensuring a line break after each subsection
}
